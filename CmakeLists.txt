######################################
#### PROJECT SETTINGS.
######################################

CMAKE_MINIMUM_REQUIRED(VERSION 2.8.3)
PROJECT(game)

# COMPILER SETTINGS.
ADD_DEFINITIONS(-Wall -pedantic -Wno-deprecated)
SET(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++11 ")

######################################
#### SET SOURCE FILES.
######################################

include(cmake/PrecompiledHeader.cmake)

if(MSVC)
  set(pch_source test-pch.cpp)
endif()

#SET(SRC main.cpp)
file(GLOB SRC src/*.cpp deps/src/glew/*.c)

ADD_EXECUTABLE(game ${SRC} ${pch_source} src/precomp.h)

add_precompiled_header(game src/precomp.h)

######################################
############ ADD  LIBRARIES
######################################

FIND_PACKAGE(OpenGL)

SET(GL_LIBS ${GLUT_LIBRARY} ${OPENGL_LIBRARY})


include_directories("deps/include")
LINK_DIRECTORIES("deps/libs")
TARGET_LINK_LIBRARIES(game glfw3.3.1 ${GL_LIBS})
